function escucharCuartoAsignado() { const q = query(collection(db, "accounts"), where("cuarto_id", "==", NUMERO_CUARTO_ASIGNADO), where("estado", "==", "abierta")); onSnapshot(q, (snapshot) => { const divVacio = document.getElementById('pantalla-vacia'); const divActivo = document.getElementById('pantalla-activa'); if (snapshot.empty) { console.log("No se encontró paciente activo para el cuarto:", NUMERO_CUARTO_ASIGNADO); pacienteActivo = null; divVacio.style.display = 'flex'; divActivo.style.display = 'none'; } else { const docSnap = snapshot.docs[0]; pacienteActivo = { id: docSnap.id, ...docSnap.data() }; // DEPURACIÓN: Esto nos dirá si realmente vienen los datos console.log("Datos recibidos del paciente:", pacienteActivo); // Aplicar datos document.getElementById('lbl-paciente').innerText = pacienteActivo.es_anonimo ? "PACIENTE ANÓNIMO" : pacienteActivo.paciente_nombre; document.getElementById('lbl-alergias').innerText = pacienteActivo.alergias || "Ninguna registrada."; // Colores const cajaAlertas = document.getElementById('caja-alertas'); cajaAlertas.className = 'alertas-box ' + (pacienteActivo.nivel_observacion || 'baja'); divVacio.style.display = 'none'; divActivo.style.display = 'flex'; } }); }
